\section{Omniprediction for single quantile level forecasts}\label{sec:single_q}

This section provides a sample efficient algorithm for omniprediction of single quantile level forecasts, both in batch and online settings. The omniprediction objective we want to minimize is given by
\begin{align*}
  \sup_{S \in S_\alpha,\, f \in \cf} \frac{1}{T} \sum_{t=1}^T \E_{p \sim \hat{P}(X_t)} [S(p, Y_t)] - \E[S(f(X_t), Y_t)]
\end{align*}
where $S_\alpha$ is the class of \textit{all} scoring functions that are consistent for the $\alpha$-level quantile functional and $\cf$ is a set of competitors.

Section \ref{subsec:decomp_single_q} discusses the previous result from \cite{ehm2016binarydecomp} on decomposition of consistent scoring rules, which is an analogous notion of propriety for single quantile forecasts. 
Using this, we introduce a new algorithm that achieves $\co(\sqrt{\log(mF)/T})$ omniprediction error bound, where $m$ is the number of discretization points and $F$ is the number of base forecasters. To do so, Section \ref{subsec:discretization} first discusses the discretization of predictions and forecasts that induces low (no) discretization error. 
Then, Section \ref{subsec:omni_single_q} states the sample efficient algorithm for omniprediction of single quantile forecasts, both in batch and online settings.


\subsection{Decomposition of 'proper' scoring rules for single quantile forecasts} \label{subsec:decomp_single_q}
Proper scoring rules incentivize forecasters to report their true predictive distribution. An analogous notion of propriety also exists for evaluating individual quantile forecasts.

Let $\cf_0$ denote the class of the probability measures on the Borel-Lebesgue sets of the real line $\R$. A function $S$ defined on $D=D_1 \times D_2 \subseteq \R^2$ is called a scoring function if $S(x,y) \geq 0$ for all $(x,y) \in D$ with $S(x,y)=0$ if $x=y$. 

Consider a functional $F \mapsto T(F) \subseteq D_1$, which can be set-valued. We define an analogous notion of propriety for single point predictions (e.g., the mean or a quantile)

\begin{defi}[Consistency]\label{def:consistency}
The scoring function $S$ is \textbf{consistent} for a functional $T$ relative to the class $\cf$ if 
\[
\E_F[S(t,Y)] \leq \E_F[S(x,Y)]
\]
for all probability measures $F \in \cf$, all $t \in T(F)$, and all point forecasts $x \in D_1$.
\end{defi}


\begin{theorem}[Section 4 in \cite{Saerens_2000}]
  Let the loss function $\ell$ satisfies the following conditions:
  \begin{enumerate}
      \item[(i)] $\ell(y, y) = 0$ for all $y \in \cy$,
      \item[(ii)] $\ell(y_1, y_2) \geq 0$ for all $y_1, y_2 \in \cy$,
      \item[(iii)] $\ell(y_1, y_2)$ is twice continuously differentiable in terms of all its arguments, except at $y_1 = y_2$ where it is simply continuous.
  \end{enumerate}
  Then, the following two statements are equivalent.
  \begin{enumerate}
      \item $\ell$ of form
          \begin{align}\label{eq:consistent-repr}
          \ell(t, y) = (\one(y<t) - \alpha)(g(t)-g(y)),
          \end{align}
          where $g$ is a differentiable function with $g'(y) > 0$ for all $y \in \cy$.
      \item $\ell$ is consistent for the $\alpha$-level quantile functional.
  \end{enumerate}
\end{theorem}
Differentiability here is not necessary for the proof. More general result without assuming smoothness is given in \cite{Thomson_1979}, page 372.
  

\paragraph{(Rephrased: From \cite{gneiting2011point})}
Up to mild regularity conditions, a scoring function $S$ is consistent for the quantile functional at level $\alpha \in (0,1)$ relative to $\cf_0$ if and only if it is of the form 
\begin{align}\label{eq:q_consistent_score}
S(x,y) = (\one(y<x)-\alpha)(g(x)-g(y)),    
\end{align}
where $g$ is non-decreasing. \todo{Note that $S$ is \textbf{right-continuous} in first and second component.}

Let $\ci$ denote the class of all \textbf{left-continuous} non-decreasing real functions, and $\cs_\alpha$ denote the class of the scoring functions $S$ of the form \eqref{eq:q_consistent_score} where $g \in \ci$.

\begin{theorem}[Theorem 1a in \cite{ehm2016binarydecomp}]
     Any member of the class $\cs_\alpha$ admits a representation of the form 
     \begin{align}\label{eq:elem_score_q}
     S(x,y) = \int_{-\infty}^{\infty} S_{\alpha,\theta}(x,y) dH(\theta),
     \end{align}
     where $H$ is a nonnegative measure and 
     \[
     S_{\alpha, \theta}(x,y) = (\one(y<x)-\alpha)(\one(\theta<x)-\one(\theta<y)) 
     =
     \begin{cases}
         1-\alpha, & y \leq \theta < x,  \\
         \alpha, & x \leq \theta < y, \\
         0, & \text{otherwise}.
     \end{cases}
     \]
     The mixing measure $H$ is unique and satisfies $dH(\theta)=dg(\theta)$ for $\theta\in\R$. We also have $H(x)-H(y) = S(x,y)/(1-\alpha)$.
\end{theorem}

\paragraph{Choquet representation} If we restrict $g\in \ci_1 \subset \ci$, where $\ci_1$ contains $g$ such that $\lim_{x\to-\infty}g(x)=0$ and $\lim_{x\to\infty}g(x)=1$, and denote the corresponding class of scoring functions as $\cs_{\alpha, 1}$, $\cs_{\alpha, \theta}$ is an extreme point of the class $\cs_{\alpha, 1}$. (If $S=(S_1+S_2)/2$ for $S_1, S_2 \in \cs_{\alpha, 1}$, then $S=S_1=S_2$)

\paragraph{Economic interpretation}
$S_{\alpha, \theta}$ corresponds to the regret compared to the best action player (bets only if $y > \theta$ and earn $\rho_G-\rho_L$, otherwise don't bet) under the following game. 
\begin{itemize}
    \item If Quinn refrains from betting, his payoff will be zero, independently of the outcome $y$.
    \item If Quinn enters the bet and $y \leq \theta$ realizes, he loses his wager, $\rho_L > 0$.
    \item If Quinn enters the bet and $y > \theta$ realizes, his winnings are $\rho_G - \rho_L$, for a gain of $\rho_G - \rho_L$.
\end{itemize}

\begin{lemma}[Discretizing all consistent losses] \label{lem:discretize_single_q}
  For any $\alpha \in (0,1) $, let $S_\alpha$ denote the class of scoring functions with representation \eqref{eq:elem_score_q} with $H([0,1])=1$.
\[
\sup_{f\in\cf, S\in S_\alpha} \E[S(p(X),Y)] - \E[S(f(X),Y)] 
=
\sup_{f\in\cf, i\in\{0,..m-1\}} \E[S_{\alpha,\theta_i}(p(X),Y)] - \E[S_{\alpha,\theta_i}(f(X),Y)] 
\]
\end{lemma}
\begin{proof}
Let $\theta_{i^*} \in \{\theta_i\}_{i=0}^{m-1}$ be the closest value  from $\theta$, where the ties are broken by rounding up. Observe that $\one(\theta < a) = \one(\theta_{i^*} < a)$ for $a \in \{\frac{j}{m}\}_{j=0}^m$. This gives 
\begin{align*}
S_{\alpha, \theta}(x,y) & = (\one(y<x)-\alpha)(\one(\theta<x)-\one(\theta<y)) \\
& = (\one(y<x)-\alpha)(\one(\theta_{i^*}<x)-\one(\theta_{i^*}<y)) = S_{\alpha, \theta_{i^*}}(x,y),
\end{align*}
for $x,y \in \{\frac{j}{m}\}_{j=0}^m$ and $\theta \in [0,1]$. Using this relationship, we obtain
\begin{align*}
& \sup_{f\in\cf, S\in S_\alpha} \E[S(p(X),Y)] - \E[S(f(X),Y)] \\
&= \sup_{f\in\cf, H:\text{pos. measure}} \E\left[\int_0^1 S_{\alpha, \theta}(p(X), Y) - S_{\alpha, \theta}(f(X),Y) dH(\theta) \right] \\
&= \sup_{f\in\cf, H:\text{pos. measure}} \int_0^1\E[S_{\alpha, \theta}(p(X), Y)] - \E[S_{\alpha, \theta}(f(X),Y)] dH(\theta)\\
&= \sup_{f\in\cf, H:\text{pos. measure}} \sum_{i=0}^{m-1} \E[S_{{\alpha, \theta_i}}(p(X), Y)] - \E[S_{{\alpha, \theta_i}}(f(X),Y)] \, 
H\left(\theta \in \left[\theta_i-\frac{1}{2m}, \theta_i +\frac{1}{2m}\right)\right) \\
&\leq \sup_{f\in\cf, i\in\{0, 1,..., m-1\}} \E[S_{{\alpha, \theta_i}}(p(X), Y)] - \E[S_{{\alpha, \theta_i}}(f(X),Y)] \, H([0,1])
\end{align*}
Note that $S_{\alpha, \theta} \equiv 0$ for $\theta \notin [0,1]$ if $X,Y \in [0,1]$. 
Since $S_{\alpha, \theta_i} \in S_\alpha$, the other direction of inequality follows, resulting in equality.
\end{proof}




\subsection{Sample efficient algorithm for omniprediction of single quantile level forecasts} \label{subsec:omni_single_q}
This subsection provides a closed form solution for the minimax program \eqref{eq:omni-twoplayer-minimax}:
\[
  \hat{P}_t(x) = \arg\min_{P \in \Delta_m} \max_{p_y \in [0,1]} 
  \mathbb{E}_{Y' \sim \text{Ber}(p_y),\, p \sim P}
  [S(p, (x, Y'); q(t))];
\]
to obtain a single quantile level forecast version of Algorithm \ref{alg:omni-twoplayer}.


Now, we denote forecasts $\{\hat{f}_{\theta_i}\}_{i=0}^{m-1}$ be the forecast that minimizes the losses $\{S_{\alpha, \theta_i}\}_{i=0}^{m-1}$. Letting $\Delta_m$ denote the set of probability distributions over $\{0, \frac{1}{m}, \frac{2}{m}, ...,1\}$, the following holds due to von Neumann's minimax theorem. Note that the expectation is linear in both $p$ and $Y$ and that $\Delta_m$ is compact convex.
\begin{lemma}\label{lem:minimax_single_q_neg}
\[
\min_{P \in \Delta_m} \max_{P_y \in \Delta_m} \sum_{i=0}^{m-1} w_i\E_{p\sim P, Y \sim P_y}[S_{\alpha,\theta_i}(p,Y) - S_{\alpha,\theta_i}(\hat{f}_{\theta_i},Y)] 
\leq 0
\]
\end{lemma}
\begin{proof}
From von Neumann's minimax theorem, we have 
\begin{align*}
\min_{P \in \Delta_m} \max_{P_y \in \Delta_m} & \sum_{i=0}^{m-1} w_i\E_{p\sim P, Y \sim P_y}[S_{\alpha,\theta_i}(p,Y) - S_{\alpha,\theta_i}(\hat{f}_{\theta_i},Y)] \\
& = \max_{P_y \in \Delta_m} \min_{P \in \Delta_m}  \sum_{i=0}^{m-1} w_i\E_{p\sim P, Y \sim P_y}[S_{\alpha,\theta_i}(p,Y) - S_{\alpha,\theta_i}(\hat{f}_{\theta_i},Y)].
\end{align*}
Since $S_{\alpha,\theta_i}$ is consistent for the quantile functional at level $\alpha$,
\[
\min_{P\in \Delta_m}\E_{p\sim P, Y \sim P_y}[S_{\alpha,\theta_i}(P,Y) - S_{\alpha,\theta_i}(\hat{f}_{\theta_i},Y)] 
\leq 
\E_{p\sim P, Y \sim P_y}[S_{\alpha,\theta_i}(\delta_{Q_\alpha(F)},Y) - S_{\alpha,\theta_i}(\hat{f}_{\theta_i},Y)] 
\leq 0.
\]
Taking $\max_{P_y \in \Delta_m}$ on both sides concludes the proof.
\end{proof}

The last component is a shortcut to compute the minimax solution by utilizing the structure of $S_{\alpha, \theta}$ losses.
\begin{align}\label{eq:minimax_q}
\hat{P}_t(x) = \arg\min_{p \in \Delta_m} \max_{Y\sim F \in \Delta_m} \sum_{i=0}^{m-1} w_i \,\E[S_{\alpha,\theta_i}(p(x),Y) - S_{\alpha,\theta_i}(\hat{f}_{\theta_i}(x),Y)] 
\end{align}


\begin{lemma} \label{lem:minimax_sol_single_q}\todo{To add special care on $k^*=m$ case} Define $B_k := \sum_{i=0}^{k-1} w_i - \sum_{i=0}^{m-1}w_i \one(\theta_i < \hat{f}_{\theta_i})$ and let $k^* = \max\{k: B_k \leq 0\}$. Note that $B_{k^*} \leq 0$ and $B_{k^*+1}>0$. Then, 
\begin{align*}
P^* = \frac{B_{k^*+1}}{B_{k^*+1} + |B_{k^*}|}\delta_{\frac{k^*}{m}} + \frac{|B_{k^*}|}{B_{k^*+1} + |B_{k^*}|}\delta_{\frac{k^*+1}{m}} 
\quad\text{and}\quad
P_y^* = \alpha \delta_0 + (1-\alpha) \delta_1 
\end{align*}
solves equation \eqref{eq:minimax_q}.
\end{lemma}
\begin{proof}
Let $O(P,P_y) := \sum_{i=0}^{m-1} w_i\E_{p\sim P, Y \sim P_y}[S_{\alpha,\theta_i}(p,Y) - S_{\alpha,\theta_i}(\hat{f}_{\theta_i},Y)]$.
It suffices to prove that $O(P^*, P_y^*) = \max_{P_y} O(P^*, P_y) = \min_P O(P, P_y^*)$. We start with decomposing $O(P, P_y)$:
\begin{align}\label{eq:oppy_q}
O(P, P_y) &= \sum_{i=0}^{m-1} w_i\E_{p\sim P, Y \sim P_y}[(\one(Y<p)-\alpha)(\one(\theta_i<p)-\one(\theta_i<Y)) - (\one(Y<\hat{f}_{\theta_i})-\alpha)(\one(\theta_i<\hat{f}_{\theta_i})-\one(\theta_i<Y))] \nonumber\\
&= \sum_{i=0}^{m-1} w_i\E_{p\sim P, Y \sim P_y}[\one(\theta_i, Y\!<\!p) - \alpha\one(\theta_i\!<\!p) - \one(\theta_i\!<\!Y\!<\!p)) - \one(\theta_i, Y\!<\!\hat{f}_{\theta_i}) + \alpha\one(\theta_i\!<\!\hat{f}_{\theta_i}) + \one(\theta_i\!<\!Y\!<\!\hat{f}_{\theta_i}))] \nonumber\\
&= \sum_{i=0}^{m-1} w_i\E_{p\sim P, Y \sim P_y}[\one(Y \leq \theta_i < p) - \alpha\one(\theta_i<p) - \one(Y \leq \theta_i < \hat{f}_{\theta_i})] + C,
\end{align}
where $C=\alpha \sum_{i=0}^{m-1} w_i \one(\theta_i < \hat{f}_{\theta_i})$ which does not depend on $P$ or $P_y$.

\smallskip
We first prove $O(P^*, P_y^*) = \max_{P_y} O(P^*, P_y)$. Let $P_y(\frac{j}{m})=p_{yj}$ for $j=0,...,m$. Then, $O(P^*, P_y)$ is rephrased as the following:
\begin{align}
O(P^*, P_y) 
=& \E_{p\sim P^*} \left[\sum_{j=0}^m p_{yj} 
\left(\sum_{i=j}^{m-1} w_i \left(\one(\theta_i < p) - \one(\theta_i < \hat{f}_{\theta_i})\right) \right) \right] + C(P^*) \nonumber\\
=& \frac{B_{k^*+1}}{B_{k^*+1} + |B_{k^*}|} \left[\sum_{j=0}^m p_{yj} 
\left(\sum_{i=j}^{k^*-1} w_i - \sum_{i=j}^{m-1} w_i \one(\theta_i < \hat{f}_{\theta_i}) \right) \right] \nonumber\\
&+ \frac{|B_{k^*}|}{B_{k^*+1} + |B_{k^*}|} \left[\sum_{j=0}^m p_{yj} 
\left(\sum_{i=j}^{k^*} w_i - \sum_{i=j}^{m-1} w_i \one(\theta_i < \hat{f}_{\theta_i}) \right) \right] + C(P^*) \label{eq:opstarpy_q},
\end{align}
where $C(P^*)$ is a value that does not depend on $P_y$ and we ignore the term hereafter.

Now observe that $A_k^j := \sum_{i=j}^{k-1} w_i - \sum_{i=j}^{m-1} w_i \one(\theta_i < \hat{f}_{\theta_i})$ is non-increasing on $j\leq k$ and non-decreasing on $k \leq j < m$. Also note that $A_k^0 \geq A_k^{m-1}$ if $k=m$ and $A_k^m = 0 \geq A_k^{m-1}$ if $k \neq m$. Therefore, $A_k^j$ achieves maximum at $j=0$ or $j=m$:
\begin{align*}
\max_{\sum_j{p_{yj}}=1}\sum_{j=0}^{m} p_{yj} \left(\sum_{i=j}^{k-1} w_i - \sum_{i=j}^{m-1} w_i \one(\theta_i < \hat{f}_{\theta_i})\right)
&= \max_{p_{y0}+p_{ym}=1} p_{y0}\left(\sum_{i=0}^{k-1} w_i - \sum_{i=0}^{m-1} w_i \one(\theta_i < \hat{f}_{\theta_i})\right) + p_{ym}\cdot 0 \\
&= \max\{A_k^0, 0\}
\end{align*}

Putting this together with equation \eqref{eq:opstarpy_q} and the fact that $B_{k^*}\leq 0$ gives
\begin{align*}
\max_{P_y}O(P^*, P_y) = p_{y0} \left(\frac{B_{k^*+1}}{B_{k^*+1} + |B_{k^*}|} \cdot B_{k*} + \frac{|B_{k^*}|}{B_{k^*+1} + |B_{k^*}|} \cdot B_{k^*+1}\right) + p_{ym}\cdot 0 = 0,
\end{align*}
which is achieved by $P_y^*$. This proves that $O(P^*, P_y^*) = \max_{P_y} O(P^*, P_y)$.

\smallskip
Next, we prove $O(P^*, P_y^*) = \min_P O(P, P_y^*)$. The decomposition \eqref{eq:oppy_q} of $O(P, P_y)$ reduces to
\begin{align*}
O(P, P_y^*) 
&= \alpha \sum_{i=0}^{m-1} w_i\E_{p\sim P}[(1-\alpha)\one(\theta_i < p) - \one(\theta_i < \hat{f}_{\theta_i})] + 
(1-\alpha) \sum_{i=0}^{m-1} w_i\E_{p\sim P}[-\alpha\one(\theta_i < p)] + C \\
&= -\alpha \sum_{i=0}^{m-1} w_i\one(\theta_i < \hat{f}_{\theta_i}) + C = 0,
\end{align*}
which does not depend on $P$. Thus, $O(P^*, P_y^*) = \min_P O(P, P_y^*)$.
\end{proof}

\begin{algorithm}
\caption{Two-player game based single quantile omniprediction (batch version)} \label{alg:omni-twoplayer_q}
\begin{algorithmic}[1]
\State \textbf{Data:} Data $\{(X_i, Y_i)\}_{i=1}^n$, hyperparameters $m \in \mathbb{N}$, $\eta > 0$, competitor functions $\{\hat{f}_{\theta_i}\}_{i=0}^{m-1}$.
\State $w_i(1) = \frac{1}{m}$, for all $i \in \{0, \ldots, m-1\}$;
\For{$t = 1, \ldots, n$}
    \State $B_k(x) := \sum_{i=0}^{k-1} w_i - \sum_{i=0}^{m-1}w_i \one(\theta_i < \hat{f}_{\theta_i}(x))$, $\quad k^*(x) = \max\{k: B_k \leq 0\}$
    \State 
    \makebox[0pt][l]{$\displaystyle
        \hat{P}_t(x) = \frac{B_{k^*(x)+1}}{B_{k^*(x)+1} + |B_{k^*(x)}|}\delta_{\frac{k^*(x)}{m}} +
        \frac{|B_{k^*(x)}|}{B_{k^*(x)+1} + |B_{k^*(x)}|}\delta_{\frac{k^*(x)+1}{m}}$
        }
    \State $w_i(t+1) \propto w_i(t) \exp\bigl(\eta \big(\mathbb{E}_{p \sim \hat{P}_t(X_t)}[S_{\theta_i}(p, Y_t)] 
        - S_{\theta_i}(\hat{f}_{\theta_i}(X_t), Y_t)\big)\bigr), \quad \forall i \in \{0, \ldots, m-1\};$
\EndFor
\State \textbf{return} $\hat{P} = \frac{1}{n} \sum_{i=1}^n \hat{P}_i$
\end{algorithmic}
\end{algorithm}




\subsubsection{Fully online algorithm for omniprediction of single quantile forecasts} 
\begin{algorithm}
  \caption{Two-player game based single quantile omniprediction (fully online version)} \label{alg:omni-twoplayer_online_q}
  \begin{algorithmic}[1]
  \State \textbf{Data:} Data stream $\{Y_t\}_{t=1}^T$, hyperparameters $m, F \in \mathbb{N}$, $\eta_w, \eta_f > 0$, competitor function class $\cf = \{f_j\}_{j=1}^{F}$.
  \State For $t=1$,$w_i(t) = \frac{1}{m}$, $\hat{f}_{\theta_i}(t) = f_1$, for all $i \in \{0, \ldots, m-1\}$;
  \State $v_{i,j}(1) = \frac{1}{F}$, for all $i \in \{0, \ldots, m-1\}$, $j \in \{1, \ldots, F\}$;
  \For{$t = 1, \ldots, T$}
    \State \textbf{1) Make prediction at time $t$}:
    \State $B_k := \sum_{i=0}^{k-1} w_i(t) - \sum_{i=0}^{m-1}w_i(t) \one(\theta_i < \hat{f}_{\theta_i}(t))$, $\quad k^* = \max\{k: B_k \leq 0\}$
      \State \textbf{Output}
      \makebox[0pt][l]{$\displaystyle
          \hat{P}_t = \frac{B_{k^*+1}}{B_{k^*+1} + |B_{k^*}|}\delta_{\frac{k^*}{m}} +
          \frac{|B_{k^*}|}{B_{k^*+1} + |B_{k^*}|}\delta_{\frac{k^*+1}{m}}$
      } \vspace{-0.1cm}\\
      \State \textbf{2) Update the weights for each $S_{\theta_i}$}:
          \State $w_i(t+1) \propto w_i(t) \exp\bigl(\eta \big(\mathbb{E}_{p \sim \hat{P}_t}[S_{\theta_i}(p, Y_t)] 
          - S_{\theta_i}(\hat{f}_{\theta_i}(t), Y_t)\big)\bigr), \quad \forall i \in \{0, \ldots, m-1\};$ \vspace{-0.1cm}\\
      \State \textbf{3) Update the best competitor function for each $S_{\theta_i}$}:
      \State $v_{i,j}(t+1) = \frac{v_{i,j}(t) \exp\bigl(-\eta_f S_{\theta_i}(\hat{f}_{j}(t), Y_t)\bigr)}{\sum_{j=1}^F v_{i,j}(t) \exp\bigl(-\eta_f S_{\theta_i}(\hat{f}_{j}(t), Y_t)\bigr)}, \quad \forall i \in \{0, \ldots, m-1\}, j \in \{1, \ldots, F\};$
      \For{$i = 0, \ldots, m-1$}
        \State Sample $j_i(t+1) \in [F] \sim v_{i,\cdot}(t+1)$;
        \State $\hat{f}_{\theta_i}(t+1) = f_{j_i(t+1)}$;
      \EndFor
  \EndFor
  \end{algorithmic}
  \end{algorithm}
  
\begin{theorem}
  \label{thm:omni-twoplayer_online_q}
  For a quantile level $\alpha \in (0,1)$, let $S_\alpha$ denote the class of scoring functions with representation \eqref{eq:elem_score_q} with $H([0,1]) \leq 1$ (the set of \textit{all} scoring functions that are consistent for the $\alpha$-level quantile functional under mild regularity conditions). Consider a class of competitors $\cf$ where each forecaster $f \in \cf$ provides $\alpha$-level quantile forecasts. With the target values $Y_t$, forecaster predictions $f(t)$ for $t \in [T], f \in \cf$ each taking values in a discretized grid of size $m$ and the learning rate $\eta_w = O(\sqrt{\frac{\log m}{T}})$ and $\eta_f = O(\sqrt{\frac{\log |\cf|}{T}})$, the output $\hat{P}_t$ from Algorithm \ref{alg:omni-twoplayer_online_q} achieves the following omniprediction error:
  \begin{align*}
  \sup_{S \in S_\alpha,\, f \in \cf} \frac{1}{T} \sum_{t=1}^T \E_{p \sim \hat{P}_t} [S(p, Y_t)] - \E[S(f(t), Y_t)]
  \leq O\left(\sqrt{\frac{\log m|\cf|}{T}}\right)
  \end{align*}
\end{theorem}


\begin{proof}
Observe that $|S_{\alpha, \theta_i}| \leq 1$. Regret bound of hedge algorithm (Theorem 1.5 of \cite{hazanoco}) gives the following:
\begin{align}
  \frac{1}{T} \sum_{t=1}^T \E_{j_i(t)}\left[S_{\alpha, \theta_i}(\hat{f}_{\theta_i}(t), Y_t) \right]
  & \,\leq\, \min_{f \in \cf} \frac{1}{T} \sum_{t=1}^T S_{\alpha, \theta_i}(f(t), Y_t) + \eta_f + \frac{\log|\cf|}{T\eta_f} \label{ieq:hedge_f_update}
\end{align}
\begin{align}
  \max_{i \in \{0, \ldots, m-1\}} \frac{1}{T} \sum_{t=1}^T 
  & \E_{p\sim \hat{P}(X_t)}\left[ S_{\alpha, \theta_i}(p, Y_t) - S_{\alpha, \theta_i}(\hat{f}_{\theta_i}(t), Y_t) \right] \nonumber\\
  & \leq \frac{1}{T} \sum_{t=1}^T \sum_{i=0}^{m-1} w_i(t) \, \E_{p\sim \hat{P}(X_t)}\left[ S_{\alpha, \theta_i}(p, Y_t) - S_{\alpha, \theta_i}(\hat{f}_{\theta_i}(t), Y_t) \right] + 4\eta_w + \frac{\log m}{T\eta_w} \label{ieq:hedge_w_update}
\end{align}

Putting these altogether, we have:
  \begin{align*}
    & \sup_{S \in S_\alpha,\, f \in \cf} \frac{1}{T} \sum_{t=1}^T \E_{p \sim \hat{P}_t} [S(p, Y_t)] - S(f(t), Y_t) \\
    & \underset{Lemma \ref{lem:discretize_single_q}}{=} \sup_{i \in \{0, \ldots, m-1\}, f \in \cf} \frac{1}{T} \sum_{t=1}^T \E_{p \sim \hat{P}_t} [S_{\alpha, \theta_i}(p, Y_t)] - S_{\alpha, \theta_i}(f(t), Y_t) \\
    & \underset{Ieq. \eqref{ieq:hedge_f_update}}{\leq} \sup_i \frac{1}{T} \sum_{t=1}^T \E_{p \sim \hat{P}_t} [S_{\alpha, \theta_i}(p, Y_t)] - \E_{j_i(t)}\left[S_{\alpha, \theta_i}(\hat{f}_{\theta_i}(t), Y_t) \right] + O\left(\sqrt{\frac{\log|\cf|}{T}}\right) \\
    & \underset{Ieq. \eqref{ieq:hedge_w_update}}{\leq} \sup_i \frac{1}{T} \sum_{t=1}^T \sum_{i=0}^{m-1} w_i(t) \, \E_{p\sim \hat{P}_t}\left[ S_{\alpha, \theta_i}(p, Y_t) - \E_{j_i(t)}\left[S_{\alpha, \theta_i}(\hat{f}_{\theta_i}(t), Y_t) \right] \right] + O\left(\sqrt{\frac{\log m|\cf|}{T}}\right) \\
    & \underset{Lemma \ref{lem:minimax_single_q_neg}}{=} O\left(\sqrt{\frac{\log m|\cf|}{T}}\right).
  \end{align*}
\end{proof}
